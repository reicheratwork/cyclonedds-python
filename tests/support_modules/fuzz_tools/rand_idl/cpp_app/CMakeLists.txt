cmake_minimum_required(VERSION 3.16)
project(xtypes_sub LANGUAGES C CXX)
find_package(CycloneDDS-CXX REQUIRED)

cmake_policy(SET CMP0060 NEW)

idlcxx_generate(TARGET xtypes_sub_lib FILES ${IDL_FILE})

# Dynamic index is generated by Python
add_executable(xtypes_sub xtypes_sub.cpp switch.cpp)
target_link_libraries(xtypes_sub xtypes_sub_lib CycloneDDS-CXX::ddscxx)

set_property(TARGET xtypes_sub PROPERTY CXX_STANDARD 17)
